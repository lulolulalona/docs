local Saving = true

local leaderstatsDataStore = game:GetService("DataStoreService"):GetDataStore("leaderstats")
local succses = pcall(function()
leaderstatsDataStore:GetAsync("TestDataStore")
end)

game.Players.PlayerAdded:Connect(function(player)
local leaderstats = Instance.new("Folder")
leaderstats.Name = "leaderstats"
leaderstats.Parent = player
--Copy
local Cash = Instance.new("NumberValue")
Cash.Name = "Cash"
Cash.Parent = leaderstats
Cash.Value = 0
--
if Saving and succses then
local Data = leaderstatsDataStore:GetAsync(player.UserId)
if Data then
for _, v in leaderstats do
v.Value = Data[v.Name]
end
end
end
end)

game.Players.PlayerRemoving:Connect(function(player)
local save = {}
for _, v in player.leaderstats:GetChildren() do
save[v.Name] = v.Value
end
if Saving and succses then
leaderstatsDataStore:SetAsync(player.UserId,save)
end
end)
